44/*!***************************************************************************

49 *
50 * @details The ADC Channels of each pixel or auxiliary channel on the device

66/*!*****************************************************************************
67 * @brief Macro to determine the pixel ADC channel number from the x-z-indices.
68 * @param x The x-index of the pixel.
69 * @param y The y-index of the pixel.
70 * @return The ADC channel number of the pixel.
71 ******************************************************************************/
72#define PIXEL_XY2CH(x, y) ((((y) << 3U) & 0x10U) | (((x) ^ 0x07U) << 1U) | ((y) & 0x01U))
73
74/*!*****************************************************************************
75 * @brief Macro to determine the pixel x-index from the ADC channel number.
76 * @param c The ADC channel number of the pixel.
77 * @return The x-index of the pixel.
78 ******************************************************************************/
79#define PIXEL_CH2X(c) ((((c) >> 1U) ^ 0x07U) & 0x07U)
80
81/*!*****************************************************************************
82 * @brief Macro to determine the pixel y-index from the ADC channel number.
83 * @param c The ADC channel number of the pixel.
84 * @return The y-index of the pixel.
85 ******************************************************************************/
86#define PIXEL_CH2Y(c) ((((c) >> 3U) & 0x02U) | ((c) & 0x01U))

89/*!*****************************************************************************

95#define PIXEL_XY2N(x, y) (((x) << 2U) | (y))

124#define PIXEL_CH2N(c) (((((c) << 1U) ^ 0x1CU) & 0x1CU) | (((c) >> 3U) & 0x02U) | ((c) & 0x01U))

239/*!*****************************************************************************
240 * @brief Converts a raw ADC channel mask to a x-y-sorted pixel mask.
241 * @param msk The raw ADC channel mask to be converted.
242 * @return The converted x-y-sorted pixel mask.
243 ******************************************************************************/
244static inline uint32_t ChannelToPixelMask(uint32_t msk)
245{
246 uint32_t res = 0;
247 for (uint_fast8_t n = 0; n < 32; n += 2)
248 {
249 res |= ((msk >> PIXEL_N2CH(n)) & 0x3U) << n;
250 }
251 return res;
252}
