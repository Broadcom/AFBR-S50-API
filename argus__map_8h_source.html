208 /*!*****************************************************************************
209  * @brief Converts a raw ADC channel mask to a x-y-sorted pixel mask.
210  * @param msk The raw ADC channel mask to be converted.
211  * @return The converted x-y-sorted pixel mask.
212  ******************************************************************************/
213 static inline uint32_t ChannelToPixelMask(uint32_t msk)
214 {
215  uint32_t res = 0;
216  for (uint_fast8_t n = 0; n < 32; n += 2)
217  {
218  res |= ((msk >> PIXEL_N2CH(n)) & 0x3U) << n;
219  }
220  return res;
221 }
