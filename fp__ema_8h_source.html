50 /*!***************************************************************************
51  * @brief Circular exponentially weighted moving average using UQ1.15 format.
52  *
53  * @details Evaluates the moving average (exponentially weighted) for circular
54  * data in UQ1.15 format.
55  * Circular data is that MAX_VALUE + 1 == MIN_VALUE. For example the
56  * usual phase information.
57  *
58  * Problem: Due to circularity of phase values, i.e. 0+x and 2PI+x are
59  * the same, the usual EMA has issues with the wrap around effect.
60  * Especially for vectors with phase around 0 (or 2PI), two values
61  * like 0 + x and PI - y are averaged to something around PI instead
62  * of 0 which would be more correct.
63  *
64  * Solution: Assume that phase jumps of more than PI are not allowed
65  * or possible. If a deviation of the new value to the smoothed signal
66  * occurs, it is clear that this stems from the wrap around effect and
67  * can be caught and correctly handled by the smoothing algorithm.
68  *
69  * Caution: If a target comes immediately into the field of view, phase
70  * jumps of > PI are indeed possible and volitional. However, the
71  * averaging break there anyway since the smoothed signal approaches
72  * only with delay to the correct values. The error made here is, that
73  * the smoothed signal approaches from the opposite direction. However,
74  * is approaches even faster since it always takes the shortest
75  * direction.
76  *
77  * @param mean The previous mean value in UQ1.15 format.
78  * @param x The current value to be added to the average UQ1.15 format.
79  * @param weight The EMA weight in UQ0.7 format.
80  * @return The new mean value in UQ1.15 format.
81  *****************************************************************************/
82 static inline uq1_15_t fp_ema15c(uq1_15_t mean, uq1_15_t x, uq0_8_t weight);

111 static inline uq1_15_t fp_ema15c(uq1_15_t mean, uq1_15_t x, uq0_8_t weight)
112 {
113  if (!weight) return x;
114  // Heeds the wrap around effect by casting dx to int16:
115  int16_t dx = (int16_t)(x - mean);
116  int32_t diff = fp_rnds(weight * dx, 8U);
117  return (uq1_15_t)(mean + diff);
118 }

128 static inline q15_16_t fp_ema16(q15_16_t mean, q15_16_t x, uq0_8_t weight)
129 {
130  if (!weight) return x;
131  if (x > mean)
132  {
133  uint32_t dx = x - mean;
134  uint32_t diff = fp_mulu(weight, dx, 8U);
135  return (q15_16_t) (mean + diff);
136  }
137  else
138  {
139  uint32_t dx = mean - x;
140  uint32_t diff = fp_mulu(weight, dx, 8U);
141  return (q15_16_t) (mean - diff);
