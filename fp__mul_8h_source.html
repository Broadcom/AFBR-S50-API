59/*!***************************************************************************
60 * @brief 64-bit implementation of an unsigned multiplication with fixed point format.
61 *
62 * @details Algorithm to evaluate a*b, where a and b are arbitrary fixed point
63 * number of 32-bit width. The multiplication is done in 64-bit and
64 * the result is shifted down by the passed shift parameter in order
65 * to return a 32-bit value.
66 * The shift is executed with correct rounding.
67 *
68 * Note that the result must fit into the 32-bit value. An assertion
69 * error occurs otherwise (or undefined behavior of no assert available).
70 *
71 * @param u The left parameter in UQx1.y1 format
72 * @param v The right parameter in UQx2.y2 format
73 * @param shift The final right shift (rounding) value.
74 * @return Result = (a*b)>>shift in UQx.(y1+y2-shift) format.
75 *****************************************************************************/
76static inline uint32_t fp_mulu(uint32_t u, uint32_t v, uint_fast8_t shift)
77{
78 assert(shift <= 32);
79#if USE_64BIT_MUL
80 uint64_t w = (uint64_t)u * (uint64_t)v;
81 return (w >> shift) + ((w >> (shift-1)) & 1U);
82#else
83 uint32_t tmp[2] = {0};
84 muldwu(tmp, u, v);
85
86 assert(shift ? tmp[0] <= (UINT32_MAX >> (32-shift)) : tmp[0] == 0);
87
88 if(32-shift)
89 return ((tmp[0] << (32-shift)) + fp_rndu(tmp[1], shift));
90 else
91 return tmp[1] > (UINT32_MAX>>1) ? tmp[0] + 1 : tmp[0];
92#endif
93}

96/*!***************************************************************************
97 * @brief 64-bit implementation of a signed multiplication with fixed point format.
98 *
99 * @details Algorithm to evaluate a*b, where a and b are arbitrary fixed point
100 * number of 32-bit width. The multiplication is done in 64-bit and
101 * the result is shifted down by the passed shift parameter in order
102 * to return a 32-bit value.
103 * The shift is executed with correct rounding.
104 *
105 * Note that the result must fit into the 32-bit value. An assertion
106 * error occurs otherwise (or undefined behavior of no assert available).
107 *
108 * @param u The left parameter in Qx1.y1 format
109 * @param v The right parameter in Qx2.y2 format
110 * @param shift The final right shift (rounding) value.
111 * @return Result = (a*b)>>shift in Qx.(y1+y2-shift) format.
112 *****************************************************************************/
113static inline int32_t fp_muls(int32_t u, int32_t v, uint_fast8_t shift)
114{
115 int_fast8_t sign = 1;
116
117 uint32_t u2, v2;
118 if(u < 0) { u2 = -u; sign = -sign; } else { u2 = u; }
119 if(v < 0) { v2 = -v; sign = -sign; } else { v2 = v; }
120
121 uint32_t res = fp_mulu(u2, v2, shift);
122
123 assert(sign > 0 ? res <= 0x7FFFFFFFU : res <= 0x80000000U);
124
125 return sign > 0 ? res : -res;
126}

129/*!***************************************************************************
130 * @brief 48-bit implementation of a unsigned multiplication with fixed point format.
131 *
132 * @details Algorithm to evaluate a*b, where a and b are arbitrary fixed point
133 * numbers with 32-bit unsigned and 16-bit unsigned format respectively.
134 * The multiplication is done in two 16x16-bit operations and the
135 * result is shifted down by the passed shift parameter in order to
136 * return a 32-bit value.
137 *
138 * Note that the result must fit into the 32-bit value. An assertion
139 * error occurs otherwise (or undefined behavior of no assert available).
140 *
141 * @param u The left parameter in Qx1.y1 format
142 * @param v The right parameter in Qx2.y2 format
143 * @param shift The final right shift (rounding) value.
144 * @return Result = (a*b)>>shift in Qx.(y1+y2-shift) format.
145 *****************************************************************************/
146static inline uint32_t fp_mul_u32_u16(uint32_t u, uint16_t v, uint_fast8_t shift)
147{
148 assert(shift <= 48);
149
150 if (shift > 16)
151 {
152 uint32_t msk = 0xFFFFU;
153 uint32_t a = (u >> 16U) * v;
154 uint32_t b = (msk & u) * v;
155 return fp_rndu(a, shift - 16) + fp_rndu(b, shift);
156 }
157 else
158 {
159 uint32_t msk = ~(0xFFFFFFFFU << shift);
160 uint32_t a = (u >> shift) * v;
161 uint32_t b = fp_rndu((msk & u) * v, shift);
162 return a + b;
163 }
164}

171 * The multiplication is done in two 16x16-bit operations and the

178 *
179 * @param u The left parameter in Qx1.y1 format
180 * @param v The right parameter in Qx2.y2 format
181 * @param shift The final right shift (rounding) value.
182 * @return Result = (a*b)>>shift in Qx.(y1+y2-shift) format.

184static inline int32_t fp_mul_s32_u16(int32_t u, uint16_t v, uint_fast8_t shift)
185{
186 return u >= 0 ? fp_mul_u32_u16(u, v, shift) : - fp_mul_u32_u16(-u, v, shift);
187}
