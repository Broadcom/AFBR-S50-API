92 static inline uint32_t fp_mul_u32_u16(uint32_t u, uint16_t v, uint_fast8_t shift)
93 {
94  assert(shift <= 48);
95 
96  if (shift > 16)
97  {
98  uint32_t msk = 0xFFFFU;
99  uint32_t a = (u >> 16U) * v;
100  uint32_t b = (msk & u) * v;
101  return fp_rndu(a, shift - 16) + fp_rndu(b, shift);
102  }
103  else
104  {
105  uint32_t msk = ~(0xFFFFFFFFU << shift);
106  uint32_t a = (u >> shift) * v;
107  uint32_t b = fp_rndu((msk & u) * v, shift);
108  return a + b;
109  }
110 }

117 static inline uint32_t fp_mulu(uint32_t u, uint32_t v, uint_fast8_t shift)
118 {
119  assert(shift <= 32);
120 
121  uint32_t tmp[2] = {0};
122  muldwu(tmp, u, v);
123 
124  assert(shift ? tmp[0] <= (UINT32_MAX >> (32-shift)) : tmp[0] == 0);
125 
126  if(32-shift)
127  return ((tmp[0] << (32-shift)) + fp_rndu(tmp[1], shift));
128  else
129  return tmp[1] > (UINT32_MAX>>1) ? tmp[0] + 1 : tmp[0];
130 }

132 static inline int32_t fp_muls(int32_t u, int32_t v, uint_fast8_t shift)
133 {
134  int_fast8_t sign = 1;
135 
136  uint32_t u2, v2;
137  if(u < 0) { u2 = -u; sign = -sign; } else { u2 = u; }
138  if(v < 0) { v2 = -v; sign = -sign; } else { v2 = v; }
139 
140  uint32_t res = fp_mulu(u2, v2, shift);
141 
142  assert(sign > 0 ? res <= 0x7FFFFFFFU : res <= 0x80000000U);
143 
144  return sign > 0 ? res : -res;
