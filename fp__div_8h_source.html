55 /*!***************************************************************************
56  * @brief 32-bit implementation of an Q15.16 division.
57  *
58  * @details Algorithm to evaluate a/b, where b is in Q15.16 format, on a 32-bit
59  * architecture with maximum precision.
60  * The result is correctly rounded and given as the input format.
61  * Division by 0 yields max. values determined by signa of numerator.
62  * Too high/low results are truncated to max/min values.
63  *
64  * Depending on the architecture, the division is implemented with a 64-bit
65  * division and shifting (Cortex-M3/4) or as a fast software algorithm
66  * (Cortex-M0/1) wich runs fast on processors without hardware division.
67  *
68  * @see https://code.google.com/archive/p/libfixmath
69  *
70  * @param a Numerator in any Qx.y format
71  * @param b Denominator in Q15.16 format
72  * @return Result = a/b in the same Qx.y format as the input parameter a.
73  *****************************************************************************/
74 static inline int32_t fp_div16(int32_t a, q15_16_t b)
75 {
76  //assert(b);
77  if (b == 0) return a < 0 ? INT32_MIN : INT32_MAX;
78 
79 #if USE_HW_DIV
80  // Tested on Cortex-M4, it takes approx. 75% of the
81  // software algorithm below.
82  int64_t c = ((int64_t) a) << 30U;
83  if ((uint32_t) (a ^ b) & 0x80000000U)
84  {
85  c = (((-c) / b) + (1 << 13U)) >> 14U;
86  if (c > 0x80000000U) return INT32_MIN;
87  return -c;
88  }
89  else
90  {
91  c = ((c / b) + (1 << 13U)) >> 14U;
92  if (c > (int64_t)INT32_MAX) return INT32_MAX;
93  return c;
94  }
95 
96 #else
97  // This uses the basic binary restoring division algorithm.
98  // It appears to be faster to do the whole division manually than
99  // trying to compose a 64-bit divide out of 32-bit divisions on
100  // platforms without hardware divide.
101  // Tested on Cortex-M0, it takes approx. 33% of the time of the
102  // 64-bit version above.
103 
104  uint32_t remainder = absval(a);
105  uint32_t divider = absval(b);
106 
107  uint32_t quotient = 0;
108  uint32_t bit = 0x10000U;
109 
110  /* The algorithm requires D >= R */
111  while (divider < remainder)
112  {
113  divider <<= 1U;
114  bit <<= 1U;
115  }
116 
117  if (!bit)
118  {
119  if ((uint32_t)(a ^ b) & 0x80000000U) // return truncated values
120  {
121  return INT32_MIN;
122  }
123  else
124  {
125  return INT32_MAX;
126  }
127  }
128 
129  if (divider & 0x80000000U)
130  {
131  // Perform one step manually to avoid overflows later.
132  // We know that divider's bottom bit is 0 here.
133  if (remainder >= divider)
134  {
135  quotient |= bit;
136  remainder -= divider;
137  }
138  divider >>= 1U;
139  bit >>= 1U;
140  }
141 
142  /* Main division loop */
143  while (bit && remainder)
144  {
145  if (remainder >= divider)
146  {
147  quotient |= bit;
148  remainder -= divider;
149  }
150  remainder <<= 1U;
151  bit >>= 1U;
152  }
153 
154  if (remainder >= divider)
155  {
156  quotient++;
157  }
158 
159  uint32_t result = quotient;
160 
161  /* Figure out the sign of result */
162  if ((uint32_t)(a ^ b) & 0x80000000U)
163  {
164  result = -result;
165  }
166 
167  return (int32_t)result;
168 #endif
169 }
