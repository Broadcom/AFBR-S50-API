65/*!***************************************************************************
66 * @brief Initializes the non-volatile memory unit and reserves a chunk of memory.
67 *
68 * @details The function is called upon API initialization sequence. If available,
69 * the non-volatile memory module reserves a chunk of memory with the
70 * provides number of bytes (size) and returns with #STATUS_OK.
71 *
72 * If not implemented, the function should return #ERROR_NOT_IMPLEMENTED
73 * in oder to inform the API to not use the NVM module.
74 *
75 * After initialization, the API calls the #NVM_Write and #NVM_Read
76 * methods to write within the reserved chunk of memory.
77 *
78 * @note The implementation of this function is optional for the correct
79 * execution of the API. If not implemented, a weak implementation
80 * within the API will be used that disables the NVM feature.
81 *
82 * @param size The required size of NVM to store all parameters.
83 * @return Returns the \link #status_t status\endlink (#STATUS_OK on success).
84 *****************************************************************************/
85status_t NVM_Init(uint32_t size);

87/*!***************************************************************************
88 * @brief Write a block of data to the non-volatile memory.
89 *
90 * @details The function is called whenever the API wants to write data into
91 * the previously reserved (#NVM_Init) memory block. The data shall
92 * be written at a given offset and with a given size.
93 *
94 * If no NVM module is available, the function can return with error
95 * #ERROR_NOT_IMPLEMENTED.
96 *
97 * @note The implementation of this function is optional for the correct
98 * execution of the API. If not implemented, a weak implementation
99 * within the API will be used that disables the NVM feature.
100 *
101 * @param offset The index offset where the first byte needs to be written.
102 * @param size The number of bytes to be written.
103 * @param buf The pointer to the data buffer with the data to be written.
104 * @return Returns the \link #status_t status\endlink (#STATUS_OK on success).
105 *****************************************************************************/
106status_t NVM_Write(uint32_t offset, uint32_t size, uint8_t const * buf);

122 * @param offset The index offset where the first byte needs to be read.
123 * @param size The number of bytes to be read.
