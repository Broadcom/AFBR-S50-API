116/*!***************************************************************************
117 * @brief Calculates the exponential of an fixed point number Q15.16 format.
118 *
119 * @details Calculates y = exp(x) in fixed point representation.
120 *
121 * Note that the result might not be 100 % accurate and might contain
122 * a small error!
123 *
124 * @see https://www.quinapalus.com/efunc.html
125 *
126 * @param x The input parameter in unsigned fixed point format Q15.16.
127 * @return Result y = exp(x) in the UQ16.16 format.
128 *****************************************************************************/
129static inline uq16_16_t fp_exp16(q15_16_t x)
130{
131 if (x == 0) return Q15_16_ONE;
132 if (x == UQ16_16_ONE) return UQ16_16_E;
133 if (x > FP_EXP16_MAX) return UQ16_16_MAX;
134 if (x < FP_EXP16_MIN) return 0;
135
136 if (x > 0)
137 {
138 if ((x << 8U) < FP_EXP24_MAX)
139 {
140 return fp_rndu(fp_exp24(x << 8U), 8U);
141 }
142 else
143 {
144 /* Use exp(2*x) = exp(x)^2 for large input number. */
145 uint32_t y = fp_exp24(x << 7U);
146 return fp_mulu(y, y, 32U); // 24 + 8
147 }
148 }
149 else
150 {
151 /* Use exp(-x) = 1/exp(x) for negative numbers. */
152 x = -x;
153
154 if ((x << 8U) < FP_EXP24_MAX)
155 {
156 uint32_t y = fp_exp24(x << 8U);
157 return fp_div16(Q15_16_ONE << 7U, fp_rndu(y, 1));
158 }
159 else
160 {
161 /* Use exp(2*x) = exp(x)^2 for large input number. */
162 uint32_t y = fp_exp24(x << 7U);
163 y = fp_div16(Q15_16_ONE << 7U, fp_rndu(y, 1));
164 return fp_mulu(y, y, 16);
165 }
166 }
167}
