113 /*!***************************************************************************

115  *
116  * @details Calculates y = exp(x) in fixed point representation.
117  *
118  * Note that the result might not be 100 % accurate and might contain

121  * @see https://www.quinapalus.com/efunc.html
122  *
123  * @param x The input parameter in unsigned fixed point format Q15.16.
124  * @return Result y = exp(x) in the UQ16.16 format.
125  *****************************************************************************/
126 static inline uq16_16_t fp_exp16(q15_16_t x)
127 {
128  if (x == 0) return Q15_16_ONE;
129  if (x == UQ16_16_ONE) return UQ16_16_E;
130  if (x > FP_EXP16_MAX) return UQ16_16_MAX;
131  if (x < FP_EXP16_MIN) return 0;
132 
133  if (x > 0)
134  {
135  if ((x << 8U) < FP_EXP24_MAX)
136  {
137  return fp_rndu(fp_exp24(x << 8U), 8U);
138  }
139  else
140  {
141  /* Use exp(2*x) = exp(x)^2 for large input number. */
142  uint32_t y = fp_exp24(x << 7U);
143  return fp_mulu(y, y, 32U); // 24 + 8
144  }
145  }
146  else
147  {
148  /* Use exp(-x) = 1/exp(x) for negative numbers. */
149  x = -x;
150 
151  if ((x << 8U) < FP_EXP24_MAX)
152  {
153  uint32_t y = fp_exp24(x << 8U);
154  return fp_div16(Q15_16_ONE << 7U, fp_rndu(y, 1));
155  }
156  else

159  uint32_t y = fp_exp24(x << 7U);
