55/*!***************************************************************************
56 * @brief Calculates the natural logarithm (base e) of an fixed point number.
57 *
58 * @details Calculates y = ln(x) = log_e(x) in fixed point representation.
59 *
60 * Note that the result might not be 100 % accurate and might contain
61 * a small error!
62 *
63 * @see https://www.quinapalus.com/efunc.html
64 *
65 * @param x The input parameter in unsigned fixed point format Q15.16.
66 * @return Result y = ln(x) in the UQ16.16 format.
67 *****************************************************************************/
68static inline q15_16_t fp_log16(uq16_16_t x)
69{
70 if (x == UQ16_16_ONE) return 0;
71 if (x == UQ16_16_E) return Q15_16_ONE;
72
73 int32_t t = 0;
74 int32_t y = 0x0A65AF68; // log(2^15) in Q7.24
75
76 /* Use ln(x/2) = ln(x)-ln(2) for large numbers. */
77 if (x > 0x80000000U)
78 {
79 x = fp_rndu(x, 1);
80 y += FP_LOG24_2;
81 }
82
83 if(x<0x00008000) x<<=16, y-=0x0B17217F;
84 if(x<0x00800000) x<<= 8, y-=0x058B90C0;
85 if(x<0x08000000) x<<= 4, y-=0x02C5C860;
86 if(x<0x20000000) x<<= 2, y-=0x0162E430;
87 if(x<0x40000000) x<<= 1, y-=0x00B17218;
88 t=x+(x>> 1); if((t&0x80000000)==0) x=t,y-=0x0067CC90;
89 t=x+(x>> 2); if((t&0x80000000)==0) x=t,y-=0x00391FF0;
90 t=x+(x>> 3); if((t&0x80000000)==0) x=t,y-=0x001E2707;
91 t=x+(x>> 4); if((t&0x80000000)==0) x=t,y-=0x000F8518;
92 t=x+(x>> 5); if((t&0x80000000)==0) x=t,y-=0x0007E0A7;
93 t=x+(x>> 6); if((t&0x80000000)==0) x=t,y-=0x0003F815;
94 t=x+(x>> 7); if((t&0x80000000)==0) x=t,y-=0x0001FE03;
95 t=x+(x>> 8); if((t&0x80000000)==0) x=t,y-=0x0000FF80;
96 t=x+(x>> 9); if((t&0x80000000)==0) x=t,y-=0x00007FE0;
97 t=x+(x>>10); if((t&0x80000000)==0) x=t,y-=0x00003FF8;
98 t=x+(x>>11); if((t&0x80000000)==0) x=t,y-=0x00001FFE;
99
100 /* Use residual to apply a final correction. */
101 x=0x80000000-x;
102 y-=x>>7;
103
104 return fp_rnds(y, 8);
105}
