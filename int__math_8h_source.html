54  * @details Finding the nearest power-of-two value s.t. |x - 2^n| becomes
55  * minimum for all n.

62 static inline uint32_t log2_round(uint32_t x)
63 {
64  if (!x) return 0;
65  uint32_t y = x;
66  uint32_t i = 0;
67  while (y >>= 1) i++;
68  return (i + ((x >> (i - 1u)) == 3u));

161 /*!***************************************************************************
162  * @brief Calculates the integer square root of x.
163  *
164  * @details The integer square root is defined as:
165  * isqrt(x) = (int)sqrt(x)
166  *
167  * @see https://en.wikipedia.org/wiki/Integer_square_root
168  * @see https://github.com/chmike/fpsqrt/blob/master/fpsqrt.c
169  *
170  * @param x Input parameter.
171  * @return isqrt(x)
172  *****************************************************************************/
173 static uint32_t isqrt(uint32_t v)
174 {
175  unsigned t, q, b, r;
176  r = v; // r = v - x²
177  b = 0x40000000; // a²
178  q = 0; // 2ax
179 
180  while( b > 0 )
181  {
182  t = q + b; // t = 2ax + a²
183  q >>= 1; // if a' = a/2, then q' = q/2
184  if( r >= t ) // if (v - x²) >= 2ax + a²
185  {
186  r -= t; // r' = (v - x²) - (2ax + a²)
187  q += b; // if x' = (x + a) then ax' = ax + a², thus q' = q' + b
188  }
189  b >>= 2; // if a' = a/2, then b' = b / 4
190  }
191  return q;
192 }
