87 static inline void muldws(int32_t w[], int32_t u, int32_t v)
88 {
89  int32_t u0, v0;
90  uint32_t u1, v1, k, t;
91  uint32_t w1, w2, w3;
92 
93  u0 = u >> 16U;
94  u1 = u & 0xFFFF;
95  v0 = v >> 16U;
96  v1 = v & 0xFFFF;
97 
98  t = u1*v1;
99  w3 = t & 0xFFFF;
100  k = t >> 16U;
101 
102  t = (uint32_t)u0*v1 + k;
103  w2 = t & 0xFFFF;
104  w1 = (uint32_t)((int32_t)t >> 16U);
105 
106  t = u1*(uint32_t)v0 + w2;
107  k = (uint32_t)((int32_t)t >> 16U);
108 
109  w[0] = u0*v0 + (int32_t)(w1 + k);
110  w[1] = (int32_t)((t << 16U) + w3);
111 /* w[1] = u*v; // Alternative. */
112 
113  return;
114 }
